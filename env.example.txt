
# Vercel & Local Environment Variables Guide
# ------------------------------------------

# [PUBLIC] - Frontend Environment Variables
# These ARE included in your client-side JavaScript bundle.
# ONLY use these for public identifiers (like Clerk's Publishable Key).
VITE_CLERK_PUBLISHABLE_KEY=pk_test_...

# [SECRET] - Backend Environment Variables
# These are NEVER sent to the browser. They live only on the server.
# Use these for API keys, secret tokens, and database credentials.

# 1. Clerk Authentication (Secret)
CLERK_SECRET_KEY=sk_test_...

# 2. Google Gemini AI (Secret)
API_KEY=AIzaSy...

# 3. Google Sheets / Service Account (Secret)
SPREADSHEET_ID=1a2b3c4d...
GOOGLE_CLIENT_EMAIL=example@project.iam.gserviceaccount.com
# Note: For GOOGLE_PRIVATE_KEY, copy the whole multi-line string with quotes
GOOGLE_PRIVATE_KEY="-----BEGIN PRIVATE KEY-----\n...\n-----END PRIVATE KEY-----\n"

# 4. Security
CRON_SECRET=generate_a_random_uuid_here
